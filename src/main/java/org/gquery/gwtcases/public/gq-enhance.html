      public void f(Element e) {

        $(".ffamily").as(Enhance).fontFamilyBox();
        $(".fsize").as(Enhance).fontSizeBox();
        $(".slider").as(Enhance).slider(0, 2);
        $(".multiselect").as(Enhance).multiselect();
        $(e).hide();
      }
    });
